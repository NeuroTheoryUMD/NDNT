113    def preprocess_weights(self, gelu_mult=2.0):
114        """self.weight is a list of the binocular weighting of each filter """       
115        # Weights should be between zeros and 1 above 1 or below zero pushed back
116        w = self.gelu(gelu_mult*self.weight) / gelu_mult
117        w = 1.0 - self.gelu(gelu_mult*(1.0-w)) / gelu_mult
118        return w

121    def forward(self, x, shift=None):
122        """
123        Propagates the input forwards through the readout
124
125        Args:
126            x: input data
127            shift (bool): shifts the location of the grid (from eye-tracking data)
128
129        Returns:
130            y: neuronal activity
131        """
132        #x = x.reshape([-1]+self.input_dims)  # 3d change -- has extra dimension at end
133        #N, c, w, h, T = x.size()   # N is number of time points -- note that its full dimensional....
134        #x = x.permute(0,1,4,2,3).reshape([N, c, w, h]) 
135        c_in, nx2 = self.input_dims[:2]
136        N = x.shape[0]
137        x = x.reshape([N, c_in, nx2])  # 3d change -- has extra dimension at end
138
139        if self.batch_sample:
140            # sample the grid_locations separately per sample per batch
141            #grid = self.sample_grid(batch_size=N, sample=self.sample)  # sample determines sampling from Gaussian
142            shifts_over_batch = self.norm_sample(batch_size=N, to_sample=self.sample)  # sample determines sampling from Gaussian
143        else:
144            # use one sampled grid_locations for all sample in the batch
145            #grid = self.sample_grid(batch_size=1, sample=self.sample).expand(N, outdims, 1, self.num_space_dims)
146            #shifts_over_batch = self.norm_sample(batch_size=1, to_sample=self.sample).expand(N, outdims, 1, self.num_space_dims)
147            shifts_over_batch = self.norm_sample(batch_size=1, to_sample=self.sample).repeat([N, 1, 1, 1])
148
149        w = self.preprocess_weights().squeeze()  # this will be a number between 0 and 1
150        Lshifts = shifts_over_batch.clone()
151        # Note: sign of shifts is such that increasing shift increases disparity (left going left, right going right)
152        Lshifts[..., 0] += (w * self.shifts * self.mult)[None,:,None]
153        Rshifts = shifts_over_batch
154        Rshifts[..., 0] += -((1-w) * self.shifts * self.mult)[None,:,None]
155
156        yL = F.grid_sample(
157            x[:,None, :,:self.NX], Lshifts, mode=self.sample_mode, align_corners=False, padding_mode='zeros')
158        yR = F.grid_sample(
159            x[:,None, :,self.NX:], Rshifts, mode=self.sample_mode, align_corners=False, padding_mode='zeros')
160        
161        y = w[None, None, :, None]*yL + (1-w)[None, None, :, None]*yR
162
163        if self.bias is not None:
164            y = y + self.bias[None, None, :, None]
165        
166        if self.NL is not None:
167            y = self.NL(y)
168        if self._num_inh > 0:
169            y = y * self._ei_mask[None, None, :, None]
170
171        self.reg.compute_activity_regularization(y)
172        return y.reshape([N, -1])
173    # END BinocShiftLayer.forward()
