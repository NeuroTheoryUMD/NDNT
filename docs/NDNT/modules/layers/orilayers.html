 49    def rotation_matrix_tensor(self, filter_dims, theta_list):
 50        """
 51        Create a rotation matrix tensor for each angle in theta_list.
 52
 53        Args:
 54            filter_dims: filter dimensions
 55            theta_list: list of angles in degrees
 56
 57        Returns:
 58            rotation_matrix_tensor: rotation matrix tensor
 59        """
 60        assert self.filter_dims[2] != 1, "OriLayer: Stimulus must be 2-D"
 61
 62        w = filter_dims[1]
 63        if w%2 == 1:             
 64            x_pos = np.repeat(np.linspace(-np.floor(w/2), np.floor(w/2), w), w)
 65            y_pos = np.tile(np.linspace(-np.floor(w/2), np.floor(w/2), w), w)
 66        else:
 67            x_pos = np.repeat((np.linspace(-(w-1), w-1, w)), w)
 68            y_pos = np.tile((np.linspace(-(w-1), w-1, w)), w)
 69
 70        indices=[]
 71        for k in range(len(theta_list)):
 72            assert theta_list[k]==int(theta_list[k]), "OriLayer: All angles must be in degrees!"
 73
 74            theta=theta_list[k]*np.pi/180
 75            rotation_matrix=rotation_matrix=np.array(
 76                [np.cos(theta), -np.sin(theta), np.sin(theta), np.cos(theta)]).reshape(2, 2)
 77            
 78            for i in range(len(x_pos)):
 79                vector = np.array([x_pos[i], y_pos[i]]).reshape(2, 1)
 80                rotated_vector = np.matmul(rotation_matrix, vector)     
 81                integer_center = np.round(rotated_vector)
 82
 83                if w%2 == 0: 
 84                    frac_part,_ =np.modf(integer_center)
 85                    for j in range(2):
 86                        if integer_center[j]%2==0:
 87                            if abs(frac_part[j])<0.5:
 88                                integer_center[j]+=np.sign(integer_center[j])
 89                            else:
 90                                integer_center[j]-=np.sign(integer_center[j])
 91
 92                x_loc = np.argwhere(x_pos == integer_center[0])
 93                if len(x_loc) == w:
 94                    x_search = x_loc.reshape(w)
 95                    y_loc = x_search[np.argwhere(y_pos[x_search] == integer_center[1])]
 96
 97                    if len(y_loc) == 0:
 98                        continue
 99                    else:
100                        indices.append([k, i, int(y_loc[0, 0])])
101                else:
102                    continue
103        
104        rotation_matrix_tensor=torch.sparse_coo_tensor(
105            torch.tensor(indices).t(), torch.ones(len(indices)), size=(len(theta_list), w**2, w**2))
106        
107        return rotation_matrix_tensor

109    def forward(self, x):
110        """
111        Forward pass through the layer.
112
113        Args:
114            x: torch.Tensor, input tensor of shape (batch_size, *input_dims)
115
116        Returns:
117            y: torch.Tensor, output tensor of shape (batch_size, *output_dims)
118        """
119        w = self.preprocess_weights() #What shape is this? (whatever self.shape is)
120        #You still need the original! w is NC*NXY*NT by NF 
121        x_0 = torch.matmul(x, w) #linearity 
122        if self.norm_type == 2:
123            x_0 = x_0 / self.weight_scale #normalization 
124
125        x_0 = x_0 + self.bias #bias 
126
127        #if self.output_norm is not None:
128        #    x = self.output_norm(x)
129
130        # Nonlinearity
131        if self.NL is not None:
132            x_0 = self.NL(x_0)
133
134        # Constrain output to be signed
135        if self._ei_mask is not None:
136            x_0 = x_0 * self._ei_mask
137        
138        x_hats = torch.zeros(tuple([x_0.shape[0], x_0.shape[1], len(self.angles)+1])) 
139        x_hats[:,:,0] = x_0 # Wrangle shapes now. 
140        w_slicing = w.reshape(self.filter_dims[0], self.filter_dims[1], self.filter_dims[2], self.filter_dims[3], self.num_filters)
141        w_flattened = w_slicing.reshape(self.filter_dims[1]*self.filter_dims[2], self.filter_dims[0]*self.filter_dims[3]*self.num_filters) 
142
143        for i in range(len(self.angles)):
144            w_theta = torch.sparse.mm(self.rotation_matrices[i], w_flattened)
145            w_reshaped = w_theta.reshape(self.shape) #NC*NXY*NT by NF 
146            x_theta = torch.sparse.mm(x, w_reshaped) #Dimensionality! (x is B by NC*NXY*NT)
147            if self.norm_type == 2:
148                x_theta = x_theta / self.weight_scale
149            x_theta = x_theta + self.bias
150            if self.NL is not None:
151                x_theta = self.NL(x_theta)
152            if self._ei_mask is not None:
153                x_theta = x_theta * self._ei_mask
154            #self.activity_regularization = self.activity_reg.regularize(x_theta) #Do we need to change that? 
155            x_hats[:,:,i+1] = x_theta
156
157        # store activity regularization to add to loss later
158        if hasattr(self.reg, 'activity_regmodule'):  # to put buffer in case old model
159            self.reg.compute_activity_regularization(x_hats)
160
161        return x_hats.reshape([x_0.shape[0], -1])

181class OriConvLayer(ConvLayer):
182    """
183    Orientation-Convolutional layer.
184
185    Will detect if needs to expand to multiple orientations (first layer, original OriConv) or use group-convolutions
186
187    2-d conv layer that creates and maintains a third convolutional dimension through grouping and weight sharing. In
188    other words, a third convolutional dimension is passed in, but the filters here act on each element of that third
189    dimension, and weight-share between different groups. 
190
191    """
192
193    def __init__(self, input_dims=None, num_filters=None, res_layer=False,
194                 filter_width=None, padding="valid", output_norm=None, angles=None, **kwargs): 
195        """
196        Initialize orientation layer.
197
198        Args:
199            input_dims: input dimensions
200            num_filters: number of filters
201            filter_width: filter spatial width -- the rest of filter dims is determined
202            angles: angles for rotation (in degrees)
203        """
204        # input validation
205        assert input_dims is not None, "OriConvLayer: Must specify input dimensions"
206        #assert len(input_dims) == 4, "OriConvLayer: Stimulus must be 2-D"
207        #assert input_dims[3] == 1, "OriConvLayer: Stimulus must be 2-D"
208        assert num_filters is not None, "OriConvLayer: Must specify number of filters"
209        assert angles is not None, "OriConvLayer: Must specify angles for rotation"
210        assert not res_layer, "OriConvLayer: res_layer not yet supported"
211
212        assert angles[0] == 0, "Angles should always start with theta=0"  # this will make calc slightly faster
213        
214        # See if gets oriented input or will need to expand
215
216        super().__init__(
217            input_dims=input_dims, num_filters=num_filters,
218            filter_dims=[input_dims[0], filter_width, filter_width, 1], padding=padding, res_layer=False,
219            output_norm=None, **kwargs)
220
221        if 'bias' in kwargs.keys():
222            assert kwargs['bias'] == False, "OriConvLayer: bias is partially implemented, but not debugged"
223
224        #self.is1D = (self.input_dims[2] == 1)
225        assert not self.is1D, "OriConvLayer: Stimulus must be 2-D"
226
227        self.angles = angles
228        NQ = len(self.angles)
229
230        self.oriented_input = input_dims[3] > 1
231        if self.oriented_input:
232            assert(input_dims[3] == NQ), "ORICONV (intermediate layer): angles input does not match stim input"
233
234        # Fix output norm to be 3d
235        if output_norm in ['batch', 'batchX']:
236            if output_norm == 'batchX':
237                affine = False
238            else:
239                affine = True
240            self.output_norm = nn.BatchNorm2d(self.num_filters*len(angles), affine=affine)
241
242        # make the ei mask and store it as a buffer,
243        # repeat it for each orientation (plus one for the original orientation)
244        if self._ei_mask is not None: 
245            self.register_buffer('_ei_mask', torch.cat((torch.ones(self.num_filters-self._num_inh), -torch.ones(self._num_inh))).repeat(NQ))
246                                
247        # Make additional window function to make filter circular
248        L = self.filter_dims[1]
249        xs = np.arange(L)+0.5-L/2
250        rs = np.sqrt(np.repeat(xs[:,None]**2, L, axis=1) + np.repeat(xs[None,:]**2, L, axis=0))
251        win_circle = np.ones([L,L], dtype=np.float32)
252        win_circle[rs > L/2] = 0.0
253        if self.window:
254            self.window_function *= torch.tensor(win_circle, dtype=torch.float32)
255        else:
256            self.register_buffer('window_function', torch.tensor(win_circle, dtype=torch.float32))
257
258        # folded_dims is num_filter * num_angles * num_incoming_filters
259        self.folded_dims = self.input_dims[0]*self.input_dims[3] 
260        # we need to set the entire output_dims so that num_outputs gets updated in the setter
261        self.output_dims = [self.output_dims[0], self.output_dims[1], self.output_dims[2], len(self.angles)]
262
263    def forward(self, x):
264        """
265        Forward pass through the layer.
266
267        Args:
268            x: torch.Tensor, input tensor of shape (batch_size, *input_dims)
269
270        Returns:
271            y: torch.Tensor, output tensor of shape (batch_size, *output_dims)
272        """
273        if self._padding == 'circular':
274            pad_type = 'circular'
275        else:
276            pad_type = 'constant'
277        
278        if self.oriented_input:
279            s = x.reshape([-1]+self.input_dims).permute([0,4,1,2,3])
280            # Set up so group conv over angles in channel dim
281            s = s.reshape([-1, self.folded_dims, self.input_dims[1], self.input_dims[2]])
282        
283        else:
284            # First conv-layer: input expanding
285            s = x.reshape(-1, self.input_dims[0], self.input_dims[1], self.input_dims[2])
286
287        w = self.preprocess_weights()
288        # permute num_filters, input_dims[0], width, height, (no lag)
289        w = w.reshape(self.filter_dims[:3]+[self.num_filters]).permute(3, 0, 1, 2)  # -> [NCXY] Note no orientation  
290
291        w_full = w.repeat(len(self.angles),1,1,1)  # (CoutxQ), Cin, XY --> each angle output is grouped
292        
293        # use torch.sparse.mm to multiply the rotation matrices by the weights
294        for ii in range(1, len(self.angles)):  # first angle always = 0 (so explicit copy -- can skip)
295            filter_range = ii*self.num_filters + np.arange(self.num_filters)
296            # rotate using torchvision transform
297            w_full[filter_range, ...] = TF.rotate(
298                img=w,
299                interpolation=torchvision.transforms.InterpolationMode.BILINEAR,
300                angle=float(self.angles[ii]))
301
302        # PAD
303        if self._fullpadding:
304            s_padded = F.pad(s, self.npads, pad_type, 0)
305            y = F.conv2d(s_padded,
306                         w_full, #rotated_ws_reshaped, 
307                         groups=self.input_dims[3],  ### ADDED
308                         bias=self.bias.repeat(len(self.angles)),
309                         stride=self.stride,
310                         dilation=self.dilation)
311        else:
312            if self.padding == 'circular':
313                s_padded = F.pad(s, self._npads, pad_type, 0)
314                y = F.conv2d(s_padded, w_full, #rotated_ws_reshaped,
315                             bias=self.bias.repeat(len(self.angles)),
316                             groups=self.input_dims[3],  ### ADDED
317                             stride=self.stride,
318                             dilation=self.dilation)
319            else: # this is faster if not circular
320                y = F.conv2d(s, w_full, #rotated_ws_reshaped,
321                             padding=(self._npads[2], self._npads[0]),
322                             groups=self.input_dims[3],  ### ADDED
323                             bias=self.bias.repeat(len(self.angles)),
324                             stride=self.stride,
325                             dilation=self.dilation)
326
327        if not self.res_layer:
328            if self.output_norm is not None:
329                y = self.output_norm(y)
330        
331        if self.NL is not None:
332            y = self.NL(y)
333            
334        if self._ei_mask is not None:
335            y = y*self._ei_mask[None, :, None, None]
336
337        # if self.res_layer:
338        #     y = y+torch.reshape(s, (-1, self.folded_dims, self.input_dims[1], self.input_dims[2]) )
339        #     if self.output_norm is not None:
340        #         y = self.output_norm(y)
341        
342        # # store activity regularization to add to loss later
343        # if hasattr(self.reg, 'activity_regmodule'):  # to put buffer in case old model
344        #     self.reg.compute_activity_regularization(y)
345
346        # pull the filters and angles apart again
347        y = y.reshape(-1, # the batch dimension
348                      len(self.angles),
349                      self.num_filters,
350                      self.output_dims[1], 
351                      self.output_dims[2])
352        # reshape y to have the orientiations in the last column
353        #y = y.permute(0, 1, 3, 4, 2) # output will be [BCWHQ]
354        y = y.permute(0, 2, 3, 4, 1) # output will be [BCWHQ]
355        # flatten the last dimensions
356        return y.reshape(-1, self.num_outputs)

510        # if hasattr(self.reg, 'activity_regmodule'):  # to put buffer in case old model

599    def hermcgen(self, mu, nu):
600        """Generate coefficients of 2D Hermite functions"""
601        nur = np.arange(nu + 1)
602        num = gamma(mu + nu + 1) * gamma(nu + 1) * ((-2) ** (nu - nur))
603        denom = gamma(mu + 1 + nur) * gamma(1 + nur) * gamma(nu + 1 - nur)
604        return num / denom

607    def hermite_2d(self, N, npts, xvalmax=None):
608        """Generate 2D Hermite function basis
609
610        Arguments:
611        N           -- the maximum rank.
612        npts        -- the number of points in x and y
613
614        Keyword arguments:
615        xvalmax     -- the maximum x and y value (default: 2.5 * sqrt(N))
616
617        Returns:
618        H           -- Basis set of size N*(N+1)/2 x npts x npts
619        desc        -- List of descriptors specifying for each
620                       basis function whether it is:
621                            'z': rotationally symmetric
622                            'r': real part of quadrature pair
623                            'i': imaginary part of quadrature pair
624
625        """
626        xvalmax = xvalmax or 2.5 * np.sqrt(N)
627        ranks = range(N)
628
629        # Gaussian envelope
630        xvalmax *= 1 - 1 / npts
631        xvals = np.linspace(-xvalmax, xvalmax, npts, endpoint=True)[...,None]
632
633        gxv = np.exp(-xvals ** 2 / 4)
634        gaussian = np.dot(gxv, gxv.T)
635
636        # Hermite polynomials
637        mu = np.array([])
638        nu = np.array([])
639        desc = []
640        for i, rank in enumerate(ranks):
641            muadd = np.sort(np.abs(np.arange(-rank, rank + 0.1, 2)))
642            mu = np.hstack([mu, muadd])
643            nu = np.hstack([nu, (rank - muadd) / 2])
644            if not (rank % 2):
645                desc.append('z')
646            desc += ['r', 'i'] * int(np.floor((rank + 1) / 2))
647
648        theta = np.arctan2(xvals, xvals.T)
649        radsq = xvals ** 2 + xvals.T ** 2
650        nbases = mu.size
651        H = np.zeros([nbases, npts, npts])
652        for i, (mui, nui, desci) in enumerate(zip(mu, nu, desc)):
653            radvals = polyval(radsq, self.hermcgen(mui, nui))
654            basis = gaussian * (radsq ** (mui / 2)) * radvals * np.exp(1j * mui * theta)
655            basis /= np.sqrt(2 ** (mui + 2 * nui) * pi * math.factorial(int(mui + nui)) * math.factorial(int(nui)))
656            if desci == 'z':
657                H[i] = basis.real / np.sqrt(2)
658            elif desci == 'r':
659                H[i] = basis.real
660            elif desci == 'i':
661                H[i] = basis.imag
662
663        # normalize
664        return torch.tensor(H / np.sqrt(np.sum(H ** 2, axis=(1, 2), keepdims=True)), dtype=torch.float32), desc, mu

608    def hermite_2d(self, N, npts, xvalmax=None):
609        """Generate 2D Hermite function basis
610
611        Arguments:
612        N           -- the maximum rank.
613        npts        -- the number of points in x and y
614
615        Keyword arguments:
616        xvalmax     -- the maximum x and y value (default: 2.5 * sqrt(N))
617
618        Returns:
619        H           -- Basis set of size N*(N+1)/2 x npts x npts
620        desc        -- List of descriptors specifying for each
621                       basis function whether it is:
622                            'z': rotationally symmetric
623                            'r': real part of quadrature pair
624                            'i': imaginary part of quadrature pair
625
626        """
627        xvalmax = xvalmax or 2.5 * np.sqrt(N)
628        ranks = range(N)
629
630        # Gaussian envelope
631        xvalmax *= 1 - 1 / npts
632        xvals = np.linspace(-xvalmax, xvalmax, npts, endpoint=True)[...,None]
633
634        gxv = np.exp(-xvals ** 2 / 4)
635        gaussian = np.dot(gxv, gxv.T)
636
637        # Hermite polynomials
638        mu = np.array([])
639        nu = np.array([])
640        desc = []
641        for i, rank in enumerate(ranks):
642            muadd = np.sort(np.abs(np.arange(-rank, rank + 0.1, 2)))
643            mu = np.hstack([mu, muadd])
644            nu = np.hstack([nu, (rank - muadd) / 2])
645            if not (rank % 2):
646                desc.append('z')
647            desc += ['r', 'i'] * int(np.floor((rank + 1) / 2))
648
649        theta = np.arctan2(xvals, xvals.T)
650        radsq = xvals ** 2 + xvals.T ** 2
651        nbases = mu.size
652        H = np.zeros([nbases, npts, npts])
653        for i, (mui, nui, desci) in enumerate(zip(mu, nu, desc)):
654            radvals = polyval(radsq, self.hermcgen(mui, nui))
655            basis = gaussian * (radsq ** (mui / 2)) * radvals * np.exp(1j * mui * theta)
656            basis /= np.sqrt(2 ** (mui + 2 * nui) * pi * math.factorial(int(mui + nui)) * math.factorial(int(nui)))
657            if desci == 'z':
658                H[i] = basis.real / np.sqrt(2)
659            elif desci == 'r':
660                H[i] = basis.real
661            elif desci == 'i':
662                H[i] = basis.imag
663
664        # normalize
665        return torch.tensor(H / np.sqrt(np.sum(H ** 2, axis=(1, 2), keepdims=True)), dtype=torch.float32), desc, mu
