  17        model, data, #val_data=None,
  18        parameters=None,

  24        history_size=100,
  25        tolerance_change=1e-7,
  26        tolerance_grad=1e-7):
  27    '''
  28    Runs fullbatch LBFGS on a Pytorch model and data dictionary
  29    Inputs:
  30        Model: Pytorch model
  31        data: Dictionary to used with Model.training_step(data)
  32    '''
  33
  34    assert isinstance(data, dict), "data must be a dictionary"
  35    model.prepare_regularization()
  36    model.train()
  37    if parameters is None:

  40        optimizer = torch.optim.LBFGS(

  44            tolerance_change=tolerance_change,
  45            line_search_fn=line_search,
  46            tolerance_grad=tolerance_grad)

  49        optimizer.zero_grad()
  50        out = model.training_step(data)

 339def create_time_embedding(stim, num_lags, up_fac=1, tent_spacing=1):
 340    """
 341    Takes a Txd stimulus matrix and creates a time-embedded matrix of size 
 342    Tx(d*L), where L is the desired number of time lags, included as last (folded) dimension. 
 343    If stim is a 3d array, its dimensions are folded into the 2nd dimension. 
 344    
 345    Assumes zero-padding.
 346     
 347    Optional up-sampling of stimulus and tent-basis representation for filter 
 348    estimation.
 349    
 350    Note that xmatrix is formatted so that adjacent time lags are adjacent 
 351    within a time-slice of the xmatrix, thus x(t, 1:nLags) gives all time lags 
 352    of the first spatial pixel at time t.
 353    
 354    Args:
 355        stim (type): simulus matrix (time must be in the first dim).
 356        num_lags
 357        up_fac (type): description
 358        tent_spacing (type): description
 359        
 360    Returns:
 361        numpy array: time-embedded stim matrix
 362        
 363    """
 364    from scipy.linalg import toeplitz
 365    # Note for myself: pdims[0] is nLags and the rest is spatial dimension
 366
 367    sz = list(np.shape(stim))
 368
 369    # If there are two spatial dims, fold them into one
 370    if len(sz) > 2:
 371        stim = np.reshape(stim, (sz[0], np.prod(sz[1:])))
 372        print('Flattening stimulus to produce design matrix.')
 373    elif len(sz) == 1:
 374        stim = stim[:, None]
 375    sz = list(np.shape(stim))
 376
 377    # No support for more than two spatial dimensions
 378    #if len(sz) > 3:
 379    #    print('More than two spatial dimensions not supported, but creating xmatrix anyways...')
 380
 381    modstim = deepcopy(stim)
 382    # Up-sample stimulus if required
 383    if up_fac > 1:
 384        # Repeats the stimulus along the time dimension
 385        modstim = np.repeat(modstim, up_fac, 0)
 386        # Since we have a new value for time dimension
 387        sz = list(np.shape(modstim))
 388
 389    # If using tent-basis representation
 390    if tent_spacing > 1:
 391        # Create a tent-basis (triangle) filter
 392        tent_filter = np.append(
 393            np.arange(1, tent_spacing) / tent_spacing,
 394            1-np.arange(tent_spacing)/tent_spacing) / tent_spacing
 395        # Apply to the stimulus
 396        filtered_stim = np.zeros(sz)
 397        for ii in range(len(tent_filter)):
 398            filtered_stim = filtered_stim + shift_mat_zpad(modstim, ii-tent_spacing+1, 0) * tent_filter[ii]
 399        modstim = filtered_stim
 400
 401    sz = list(np.shape(modstim))
 402    lag_spacing = tent_spacing
 403
 404    # If tent_spacing is not given in input then manually put lag_spacing = 1
 405    # For temporal-only stimuli (this method can be faster if you're not using
 406    # tent-basis rep)
 407    # For myself, add: & tent_spacing is empty (= & isempty...).
 408    # Since isempty(tent_spa...) is equivalent to its value being 1 I added
 409    # this condition to the if below temporarily:
 410    if sz[1] == 1 and tent_spacing == 1:
 411        xmat = toeplitz(np.reshape(modstim, (1, sz[0])),
 412                        np.concatenate((modstim[0], np.zeros(num_lags - 1)),
 413                                       axis=0))
 414    else:  # Otherwise loop over lags and manually shift the stim matrix
 415        xmat = np.zeros([sz[0], sz[1], num_lags])
 416        for lag in range(num_lags):
 417            for xx in range(sz[1]):
 418                xmat[:, xx, lag] = shift_mat_zpad(modstim[:, xx], lag_spacing * lag, 0)
 419        xmat = xmat.reshape([sz[0], -1])
 420
 421    return xmat
 422# END create_time_embedding

 532def design_matrix_tent_basis( s, anchors, zero_left=False, zero_right=False):
 533    """Produce a design matrix based on continuous data (s) and anchor points for a tent_basis.
 534    Here s is a continuous variable (e.g., a stimulus) that is function of time -- single dimension --
 535    and this will generate apply a tent basis set to s with a basis variable for each anchor point. 
 536    The end anchor points will be one-sided, but these can be dropped by changing "zero_left" and/or
 537    "zero_right" into "True".
 538
 539    Inputs: 
 540        s: continuous one-dimensional variable with NT time points
 541        anchors: list or array of anchor points for tent-basis set
 542        zero_left, zero_right: boolean whether to drop the edge bases (default for both is False)
 543    Outputs:
 544        X: design matrix that will be NT x the number of anchors left after zeroing out left and right
 545    """
 546
 547    if len(s.shape) > 1:
 548        assert s.shape[1] == 1, 'Can only work on 1-d variables currently'
 549        s = np.squeeze(s)
 550
 551    NT = len(s)
 552    NA = len(anchors)
 553    X = np.zeros([NT, NA])
 554    for nn in range(NA):
 555        if nn == 0:
 556            #X[np.where(s < anchors[0])[0], 0] = 1
 557            X[:, 0] = 1
 558        else:
 559            dx = anchors[nn]-anchors[nn-1]
 560            X[:, nn] = np.minimum(np.maximum(np.divide( deepcopy(s)-anchors[nn-1], dx ), 0), 1)
 561        if nn < NA-1:
 562            dx = anchors[nn+1]-anchors[nn]
 563            X[:, nn] *= np.maximum(np.minimum(np.divide(np.add(-deepcopy(s), anchors[nn+1]), dx), 1), 0)
 564    if zero_left:
 565        X = X[:,1:]
 566    if zero_right:
 567        X = X[:,:-1]
 568    return X

 571def tent_basis_generate( xs=None, num_params=None, doubling_time=None, init_spacing=1, first_lag=0 ):
 572    """Computes tent-bases over the range of 'xs', with center points at each value of 'xs'.
 573    Alternatively (if xs=None), will generate a list with init_space and doubling_time up to
 574    the total number of parameters. Must specify xs OR num_params. 
 575    Note this assumes discrete (binned) variables to be acted on.
 576    
 577    Defaults:
 578        doubling_time = num_params
 579        init_space = 1"""
 580
 581    # Determine anchor-points
 582    if xs is not None:
 583        tbx = np.array(xs,dtype='int32')
 584        if num_params is not None: 
 585            print( 'Warning: will only use xs input -- num_params is ignored.' )
 586    else:
 587        assert num_params is not None, 'Need to specify either xs or num_params'
 588        if doubling_time is None:
 589            doubling_time = num_params+1  # never doubles
 590        tbx = np.zeros( num_params, dtype='int32' )
 591        cur_loc, cur_spacing, sp_count = first_lag, init_spacing, 0
 592        for nn in range(num_params):
 593            tbx[nn] = cur_loc
 594            cur_loc += cur_spacing
 595            sp_count += 1
 596            if sp_count == doubling_time:
 597                sp_count = 0
 598                cur_spacing *= 2
 599
 600    # Generate tent-basis given anchor points
 601    NB = len(tbx)
 602    NX = (np.max(tbx)+1).astype(int)
 603    tent_basis = np.zeros([NX,NB], dtype='float32')
 604    for nn in range(NB):
 605        if nn > 0:
 606            dx = tbx[nn]-tbx[nn-1]
 607            tent_basis[range(tbx[nn-1], tbx[nn]+1), nn] = np.array(list(range(dx+1)))/dx
 608        elif tbx[0] > 0:  # option to have function go to zero at beginning
 609            dx = tbx[0]
 610            tent_basis[range(tbx[nn]+1), nn] = np.array(list(range(dx+1)))/dx
 611        if nn < NB-1:
 612            dx = tbx[nn+1]-tbx[nn]
 613            tent_basis[range(tbx[nn], tbx[nn+1]+1), nn] = 1-np.array(list(range(dx+1)))/dx
 614
 615    return tent_basis

 618def shift_mat_zpad(x, shift, dim=0):
 619    """Takes a vector or matrix and shifts it along dimension dim by amount 
 620    shift using zero-padding. Positive shifts move the matrix right or down.
 621    
 622    Args:
 623        x (type): description
 624        shift (type): description
 625        dim (type): description
 626        
 627    Returns:
 628        type: description
 629            
 630    Raises:
 631            
 632    """
 633
 634    assert x.ndim < 3, 'only works in 2 dims or less at the moment.'
 635    if x.ndim == 1:
 636        oneDarray = True
 637        xcopy = np.zeros([len(x), 1])
 638        xcopy[:, 0] = x
 639    else:
 640        xcopy = deepcopy(x)
 641        oneDarray = False
 642    sz = list(np.shape(xcopy))
 643
 644    if sz[0] == 1:
 645        dim = 1
 646
 647    if dim == 0:
 648        if shift >= 0:
 649            a = np.zeros((shift, sz[1]))
 650            b = xcopy[0:sz[0]-shift, :]
 651            xshifted = np.concatenate((a, b), axis=dim)
 652        else:
 653            a = np.zeros((-shift, sz[1]))
 654            b = xcopy[-shift:, :]
 655            xshifted = np.concatenate((b, a), axis=dim)
 656    elif dim == 1:
 657        if shift >= 0:
 658            a = np.zeros((sz[0], shift))
 659            b = xcopy[:, 0:sz[1]-shift]
 660            xshifted = np.concatenate((a, b), axis=dim)
 661        else:
 662            a = np.zeros((sz[0], -shift))
 663            b = xcopy[:, -shift:]
 664            xshifted = np.concatenate((b, a), axis=dim)
 665
 666    # If the shift in one direction is bigger than the size of the stimulus in
 667    # that direction return a zero matrix
 668    if (dim == 0 and abs(shift) > sz[0]) or (dim == 1 and abs(shift) > sz[1]):
 669        xshifted = np.zeros(sz)
 670
 671    # Make into single-dimension if it started that way
 672    if oneDarray:
 673        xshifted = xshifted[:,0]
 674
 675    return xshifted

 678def generate_xv_folds(nt, num_folds=5, num_blocks=3, which_fold=None):
 679    """Will generate unique and cross-validation indices, but subsample in each block
 680        NT = number of time steps
 681        num_folds = fraction of data (1/fold) to set aside for cross-validation
 682        which_fold = which fraction of data to set aside for cross-validation (default: middle of each block)
 683        num_blocks = how many blocks to sample fold validation from"""
 684
 685    test_inds = []
 686    NTblock = np.floor(nt/num_blocks).astype(int)
 687    block_sizes = np.zeros(num_blocks, dtype='int32')
 688    block_sizes[range(num_blocks-1)] = NTblock
 689    block_sizes[num_blocks-1] = nt-(num_blocks-1)*NTblock
 690
 691    if which_fold is None:
 692        which_fold = num_folds//2
 693    else:
 694        assert which_fold < num_folds, 'Must choose XV fold within num_folds =' + str(num_folds)
 695
 696    # Pick XV indices for each block
 697    cnt = 0
 698    for bb in range(num_blocks):
 699        tstart = np.floor(block_sizes[bb] * (which_fold / num_folds))
 700        if which_fold < num_folds-1:
 701            tstop = np.floor(block_sizes[bb] * ((which_fold+1) / num_folds))
 702        else: 
 703            tstop = block_sizes[bb]
 704
 705        test_inds = test_inds + list(range(int(cnt+tstart), int(cnt+tstop)))
 706        cnt = cnt + block_sizes[bb]
 707
 708    test_inds = np.array(test_inds, dtype='int')
 709    train_inds = np.setdiff1d(np.arange(0, nt, 1), test_inds)
 710
 711    return train_inds, test_inds

 714def initialize_gaussian_envelope( ws, w_shape):
 715    """
 716    This assumes a set of filters is passed in, and windows by Gaussian along each non-singleton dimension
 717    ws is all filters (ndims x nfilters)
 718    wshape is individual filter shape
 719    """
 720    ndims, nfilt = ws.shape
 721    assert np.prod(w_shape) == ndims
 722    wx = np.reshape(deepcopy(ws), w_shape + [nfilt])
 723    for dd in range(1,len(w_shape)):
 724        if w_shape[dd] > 1:
 725            L = w_shape[dd]
 726            genv = np.exp(-(np.arange(L)-L/2)**2/(2*(L/6)**2))
 727            if dd == 0:
 728                wx = np.einsum('abcde, a->abcde', wx, genv)
 729            elif dd == 1:
 730                wx = np.einsum('abcde, b->abcde', wx, genv)
 731            elif dd == 2:
 732                wx = np.einsum('abcde, c->abcde', wx, genv)
 733            else:
 734                wx = np.einsum('abcde, d->abcde', wx, genv)
 735    return np.reshape(wx, [-1, nfilt])

 739def pixel2mu( p, L=60, flip_axes=True ):
 740    """
 741    Converts from pixel coordinates to mu values, used by grid_sample. The default will flip
 742    horizontal and vertical axes automatically, which is what grid_sample requires
 743    Pixels are starting with number 0 up to L-1, converted to range of -1 to 1
 744    
 745    Args:
 746        p: list of pixel locations, presumably num_locations x 2 (horizontal and vertical coords)

 749    
 750    Returns:
 751        x: mu values
 752    """
 753    x = (2*p+1)/L - 1
 754    if flip_axes:
 755        if len(x.shape) < 2:
 756            print("Warning: cannot flip axes, since only passed in one dimension")
 757        else:
 758            assert x.shape[1] == 2, "need 2 axes to flip them"
 759            x = np.roll(x, 1, axis=1)
 760    return x

 764def mu2pixel( x, L=60, force_int=True, enforce_bounds=True, flip_axes=True ):
 765    """
 766    Converts from mu values back into pixel coordinates, where mus are coordinates used by grid_sample.
 767    Can be continuous (fractional pixels), but default is rounding to nearest int
 768
 769    Args: 
 770        x: mu values, assuming num_locations x 2 (although can be 1-dimensional)
 771        L: size of grid (number of pixels), assuming square (default 60)
 772        force_int: force to be integer (versus continuous-valued)
 773        enforce bounds: go to edges if answer ends up bigger than L-1 or smaller than 0
 774        flip_axes: whether to swap horizontal and vertical axes as grid_sample needs (default: True)
 775
 776    Returns:
 777        p: pixel values
 778    """

 780        if len(x.shape) < 2:

 783            assert x.shape[1] == 2, "need 2 axes to flip them"

 788        a = np.where(p > L-1)[0]
 789        if len(a) > 0:   
 790            p[a] = L-1
 791        a = np.where(p < 0)[0]

 793            p[a] = 0

 796            return np.round(p).astype(np.int64)
 797        else:
 798            return int(np.round(p))
 799    else:
 800        return p
